-}} {{- end -}} {{- end -}} {{- end -}} {{- $domain := $url | replaceRE `^https?://([^/]+).*` "$1" -}} {{- if not $title -}} {{- $title = $domain -}} {{- end -}} {{/* Qiita の場合、画像 URL の & を & に置換 */}}
